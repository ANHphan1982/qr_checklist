function StepDot({ state }) {
  // state: "done" | "active" | "todo"
  if (state === "done") {
    return (
      <div className="w-8 h-8 rounded-full bg-indigo-500 flex items-center justify-center
                      shadow-sm shadow-indigo-500/30">
        <IconCheck className="w-4 h-4 text-white"/>
      </div>
    );
  }
  if (state === "active") {
    return (
      <div className="w-8 h-8 rounded-full bg-indigo-600 flex items-center justify-center
                      shadow-md shadow-indigo-600/40 ring-4 ring-indigo-600/20">
        <div className="w-3 h-3 rounded-full bg-white"/>
      </div>
    );
  }
  return (
    <div className="w-8 h-8 rounded-full bg-slate-200 dark:bg-slate-700
                    flex items-center justify-center"/>
  );
}

// Stepper — horizontal progress indicator
// ---------------------------------------------------------------------------
function Stepper({ current, total }) {
  return (
    <div className="flex items-center gap-1.5">
      {Array.from({ length: total }).map((_, i) => {
        const n = i + 1;
        const state = n < current ? "done" : n === current ? "active" : "todo";
        return (
          <div key={n} className="flex items-center gap-1.5">
            <StepDot state={state}/>
            {n < total && (
              <div className={`h-0.5 w-4 rounded-full transition-colors
                ${n < current ? "bg-indigo-500" : "bg-slate-200 dark:bg-slate-700"}`}/>
            )}
          </div>
        );
      })}
    </div>
  );
}
